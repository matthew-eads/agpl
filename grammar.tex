\paragraph{AGPL Syntax}
\begin{grammar}
<game> ::= `Gamestate:{' [<gamestate>] `}'\\
           `Player:{' <Dec> `}'\\
           `Move:{' <Dec> `}'\\
           `isVailid:{' <Exp> `}'\\
           `possMoves:{' <Exp> `}'\\
           `outcome:{' <Exp> `}'\\
           `initialState:{' <InitState> `}'\\
           `fromString:{' <Exp> `}'\\
           `$' [<Dec>] `$'
           

%% <gamestatelist> ::= <gamestate> <gamestatelist>
%% \alt <empty>


<gamestate> ::= `Board:{' <Dec>
\alt `Piece:{' <Dec>
\alt `Hand:{' <Dec>
\alt `Turn:{' <Dec>
\alt <string>`:{' <Dec>

<InitState> ::= `Board:{' <Exp> `}'\\
                `Turn:{' <Exp> `}'

<Exp> ::= <Template Haskell Expression>

<Dec> ::= <Template Haskell Declaration>

\end{grammar}
